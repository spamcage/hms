代码: 全选

return array(
	
	'sieve' => false,
	'sieve.autoresponder' => true,
	'sieve.forward' => true,
	'sieve.filters' => true,
	'sieve.config.host' => '',
	'sieve.config.port' => 2000,
	'sieve.config.filters-folder-charset' => 'utf-8', // [utf7-imap, utf-8]

	'plugins.external-services' => true,

	),
	'plugins.hmailserver-change-password' => true,
	'plugins.hmailserver-change-password.config.login' => 'administrator',
